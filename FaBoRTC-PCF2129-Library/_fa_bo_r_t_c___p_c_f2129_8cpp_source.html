178 DateTime FaBoRTC_PCF2129::now(void) {
179  Wire.beginTransmission(_i2caddr);
180  Wire.write(PCF2129_SECONDS);
181  Wire.endTransmission();
182  Wire.requestFrom(_i2caddr, (uint8_t)7);
183  while(!Wire.available());
184  uint8_t seconds = bcdToDec(Wire.read());
185  uint8_t minutes = bcdToDec(Wire.read());
186  uint8_t hours = bcdToDec(Wire.read());
187  uint8_t days = bcdToDec(Wire.read());
188  Wire.read(); // blank read weekdays
189  uint8_t months = bcdToDec(Wire.read());
190  uint16_t years = bcdToDec(Wire.read()) +2000;
191 
192  return DateTime(years, months, days, hours, minutes, seconds);
193 }
